01

Հայտարարում և Մուտք

Մասիվները օբյեկտների հատուկ տեսակ են, որոնք նախատեսված են կարգավորված տվյալներ պահելու համար:

Ստեղծում (Literal)

Ամենատարածված ձևը քառակուսի փակագծերի օգտագործումն է:

let fruits = [ "🍎", "🍊", "🍐" ]; // Հագեցած մասիվ
let empty = [ ]; // Դատարկ մասիվ

Տարրերի հասանելիություն

Զանգվածի տարրերը համարակալվում են 0-ից: Ցանկացած տարրի կարելի է դիմել իր համապատասխան ինդեքսի միջոցով:

fruits[0]; // "🍎" (Առաջին տարր)
fruits[1]; // "🍊" (Երկրորդ տարր)

Տեսակների ազատություն

Մասիվը կարող է պահել ցանկացած տեսակի տվյալ՝ string, number, boolean, object և նույնիսկ այլ մասիվներ:

let mixed = [ 1, "🍎", true, {id: 1} ];

Տիպի ճշգրիտ ստուգում

console.log(typeof [ ]); // "object" (Ոչ բավարար)
Array.isArray([ ]); // true (Հուսալի)

* Քանի որ JS-ում մասիվը օբյեկտ է, typeof-ը միշտ կվերադարձնի "object": Ճիշտ ստուգման համար միշտ օգտագործեք Array.isArray():

Մասիվը հիշողության մեջ (Heap) պահվում է որպես օբյեկտ՝ հատուկ ներքին կառուցվածքով, որտեղ length-ը ավտոմատ թարմացվող հատկություն է:

02

Հիմնական Մեթոդներ (LIFO/FIFO)

Վերջից (Արագ)

  • push: Ավելացնում է տարր վերջից.
  • pop: Հեռացնում է վերջին տարրը.

Սկզբից (Դանդաղ)

  • unshift: Ավելացնում է տարր սկզբից.
  • shift: Հեռացնում է առաջին տարրը.

Կոդի Օրինակ (Queue & Stack)

let queue = ["Apple", "Orange"];

// FIFO (First In, First Out) - Հերթ
queue.push("Pear"); // ["Apple", "Orange", "Pear"]
queue.shift(); // հեռացնում է "Apple" -> ["Orange", "Pear"]

let stack = [];
stack.push(1); // [1]
stack.push(2); // [1, 2]
stack.pop(); // հեռացնում է 2 -> [1]

Մասիվների Շրջանցում

for...of ցիկլ

for (let fruit of fruits) {
  console.log(fruit);
}

Ամենապարզ և ժամանակակից ձևը արժեքները ստանալու համար:

length հատկությունը
