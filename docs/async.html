02

Ինչպե՞ս է դա աշխատում

Վիզուալիզացիա և Օգտագործում

// Պատկերացնենք տվյալների ստացում սերվերից
async function getUserData() {
  try {
    console.log("Հարցումն ուղարկված է...");

    // Ծրագիրը սպասում է այստեղ, մինչև fetch-ը կավարտվի
    let response = await fetch('https://api.example.com/user');

    // Սպասում է, մինչև JSON-ը պատրաստ լինի
    let data = await response.json();

    console.log("Տվյալները ստացված են:", data);
  } catch (error) {
    console.error("Սխալ տեղի ունեցավ:", error);
  }
}

03

Համեմատություն

Promises vs Async/Await

Promises (.then)

function load() {
  fetch('/api')
    .then(res => res.json())
    .then(data => console.log(data));
}

Async / Await

async function load() {
  const res = await fetch('/api');
  const data = await res.json();
  console.log(data);
}
